\section{Constantes}
\begin{description}
\item aux dist(p1: GPS, p2: GPS) : Dist
\item \pred{esUnViajeValido}{\In v: Viaje}{
		(\forall i : \ent)(0 \leq i < \longitud v \wedge 0 \leq j < \longitud{v}-1  \implicaLuego gpsEnRango (v[j]_1,v[j+1]_1)) \wedge tiempoEnRango(v[i]_{0})
    }

\item \pred{gpsEnRango}{\In g1: GPS, \In g2: GPS}{
		( (-90.0 \leq g1_{0},g2_{0} \leq 90.0) \y (-180.0 \leq g1_{1},g2_{1} \leq 180.0) ) \y (  dist(g1,g2) \approx 20 )
    }
    
\item \pred{tiempoEnRango}{\In t: Viaje} {
    t_{0} > 0
    }

\item \pred{esUnRecorridoValido}{\In r: Recorrido}{
		(\forall j:\ent)(0 \leq j < \longitud{r}-1  \implicaLuego gpsEnRango (r[j],r[j+1]))
		}

\item DistanciaMaximaMetros = 444
    
\end{description}

\begin{proc}{construirGrilla}{\In esq1: GPS, \In esq2: GPS, \In n:\ent, \In m:\ent, \out g: Grilla}{}
    \pre{n\geq 1 \wedge m \geq 1 \wedge gpsEnRango(esq1) \wedge gpsEnrango(esq2) \wedge ( (esq1_0 > esq2_0) \wedge (esq1_1 < esq2_1}
    \post{ todasLasFilas (g) \wedge todasLasColumnas (g) \wedge enRangoDeGrilla (g, esq1, esq2) \wedge mismoTamañoDeRectángulos (g)}
    
    \pred {todasLasFilas} {g:Grilla} {
    ((\forall i:\ent)(\exists j:\ent)(0 < i \leq n \implicaLuego g[j]_2_0 = i))
    }
    
    \pred {todasLasColumnas} {g:Grilla} {
    ((\forall i:\ent)(\exists j:\ent)(0 < i \leq m \implicaLuego g[j]_2_1 = i))
    }
    
    \pred {enRangoDeGrilla} {g:Grilla, esq1: GPS, esq2: GPS}{
    ((\forall i:\ent)(0 \leq j < |g| \implicaLuego (esq2_0 \leq g[2]_j_0 \leq esq1_0) \wedge (esq2_1 \leq g[2]_j_1 \leq esq1_1)))
    }
    
    \pred {mismoTamañoDeRectangulos} {g: Grilla} {
    (\forall x:\ent) ¬(\exists y:\ent) (0 \leq x,y < |g| \implicaLuego ((base(g[x]) \neq base(g[y])) \wedge (altura(g[x]) \neq altura(g[y])))
    }
    
    \aux {base}{in g:Grilla}{\float\\}
    {g_1_1 - g_0_1}
    
    \aux {altura}{in g:Grilla}{\float\\}
    {g_1_0 - g_0_0}
    
\end{proc}

\subsection*{Ejercicio 12}
\begin{proc}{corregirViaje}{\Inout v:Viaje,\In errores: \TLista{Tiempo}}{}
	\pre{\longitud{v} > 5 \y esUnViajeValido(v) \y \longitud{errores} > 0.1 \times \longitud{v} \y v = v_0 }
	\post{(\forall i: \ent)(0 \leq i < \longitud{v} \implicaLuego v[i]_0 \in errores \y
	(\exists k,j :\ent)(0 \leq k,j < \longitud{v} \y k \neq j \yLuego
	masCercanos(v[i]_0, v[k]_0,v) \y masCercanos(v[i]_0, v[j]_0,v) \yLuego
	estaEnRectayPunto(v[i],v[k],v[j])))
	}
	
	\pred{masCercanos}{i,j:Tiempo,v:Viaje}{
		(\forall r:\ent)(0 \leq r < \longitud{v} \implicaLuego restaModulo(i,v[r]_0) \geq restaModulo(i,j))
		}
	

	\pred{estaEnRectayPunto}{g,p1,p2: (Tiempo X Gps)}{
		estaEnRectaLat(g,p1,p2) \y estaEnRectaLong(g,p1,p2)
		}

	\pred{estaEnRectaLat}{g,p1,p2: (Tiempo X Gps)}{
	    g_0_0 = \frac{(p2_0_0-p1_0_0)(g_2-p1_2)}{p2_2-p1_2} + p1_0_0
		}
	\pred{estaEnRectaLong}{g,p1,p2: GPS}{
        g_0_1 = \frac{(p2_0_1-p1_0_1)(g_2-p1_2)}{p2_2-p1_2} + p1_0_1
		}

		\aux{restaModulo}{a,b: \ent}{\ent \\}{
			\IfThenElse{a \geq b}{a-b}{b-a}
			}
\end{proc}

\subsection*{Ejercicio 13}
\begin{proc}{histograma}{\In xs: \TLista{Viaje},\In bins: \ent,out cuentas: \TLista{\ent},out limites: \TLista{\float}}{}
	\pre{(\forall i: \ent)(0 \leq i < \longitud{xs} \implicaLuego esUnViajeValido(xs[i])) \wedge bins \geq 1}
	\post{limitesValidos(limites,bins,xs) \y cuentasValidas(cuentas,limites,xs,bins)}
	\pred{limitesValidos}{\In limites: \TLista{\float}, \In bins: \ent, \In xs: \TLista{Viaje}}{
		\longitud{limites} = bins + 1 \y \\
		(\forall i: \ent)(0 \leq i < \longitud{limites}-1  \y  limites[i] > limites[i+1])  \y \\ 
		(\exists k,j: \float)(velMayoryMenordeMaximas(k,j) \yLuego (\forall k: \ent)(0 \leq k < \longitud{limites}-1 \implicaLuego limites[i] = limites[i+1] - \frac{k - j}{bins})
		}
	\pred{cuentasValidas}{\In cuentas: \TLista{\ent},\In limites: \TLista{\float},\In xs: \TLista{Viaje},\In bins:\ent}{
		\longitud{cuentas} = bins \y \\
		(\forall i: \ent)(0 \leq i < \longitud{cuentas} %que es igual a \longitud{limites} -1 
		\implicaLuego 
		cuentas[i] = \sum_{j=0}^{\longitud{xs}-1} \IfThenElse{(\exists k: \ent)(esVelocidadMaxima(k,xs[j]) \yLuego limites[i] \leq  k \leq limites[i+1])}{1}{0})
		}
	\pred{esVelocidadMaxima}{\In i: \ent,\In v: Viaje}{
		(\forall j: \ent)(0 \leq j < \longitud v \implicaLuego i \geq \frac{dist(v[j]_1,distanciaSiguiente(v[j]_1,v))}{20})
		}
	\pred{velMayoryMenordeMaximas}{\In velMayorMaxima, \In velMenorMaxima,\In limites: \TLista{\float},\In bins: \ent}{
		(\forall viajes:\ent)(\exists velMax: \ent)(0 \leq viajes < \longitud{xs} \implicaLuego esVelocidadMaxima(velMax,xs[j]) \yLuego
		(velMayorMaxima \geq velMax \wedge velMenorMaxima \leq velMax)
	}
\end{proc}
